CREATE DATABASE BDSeleccionFut

CREATE TABLE usuario (
    idusu INT AUTO_INCREMENT NOT NULL,
    codigo VARCHAR(75) NULL,
    email VARCHAR(150) NULL,
    password VARCHAR(70) NULL,
    activo BOOL NULL,
    CONSTRAINT users_pk PRIMARY KEY (idusu)
);


CREATE TABLE rol (
    idrol INT AUTO_INCREMENT NOT NULL,
    rolusu VARCHAR(120) NULL,
    CONSTRAINT roles_pk PRIMARY KEY (idrol)
);


CREATE TABLE rol_de_usuarios (
    idusu INT NOT NULL,
    idrol INT NOT NULL,
    CONSTRAINT user_role_pk PRIMARY KEY (idusu, idrol),
    CONSTRAINT user_role_FK FOREIGN KEY (idusu) REFERENCES usuario(idusu),
    CONSTRAINT user_role_FK_1 FOREIGN KEY (idrol) REFERENCES rol(idrol)
);
